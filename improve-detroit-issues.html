# custom function to transpose while preserving names
transpose_df <- function(df) {
  t_df <- data.table::transpose(df)
#   colnames(t_df) <- rownames(df)
  colnames(t_df) <- c("Observation")
  # rownames(t_df) <- colnames(df)
  rownames(t_df) <- colname_desc
  t_df <- t_df %>%
    tibble::rownames_to_column(.data = ., var = "Dataset Dimension") %>%
    tibble::as_tibble(.)
  return(t_df)
}

Number of issues over time

Based on the “Created At” field, the relative number of issues per month from July 14 2014 to Feb 22 2024.

  • Min: 2014-07-14 12:54:47
  • Mean: 2020-03-19 09:37:07.680974
  • Max: 2024-02-22 05:27:27
Code 
df_time <- improve_detroit_issues_all |> 
  select(Created_At) |> #, Acknowledged_At, Closed_At, Reopened_At, Updated_At)
  mutate(created_yr_month = as.Date(as.yearmon(Created_At))) |>
  group_by(created_yr_month) |>
  summarize(total = n())


ggplot(df_time, aes(x=created_yr_month, y=total, group=1)) +
  geom_point(color = "#279989") +
  geom_ma(ma_fun = SMA, n = 3, color = "#004445", linetype = 1, size = 1.25) +
  theme_minimal() +
  labs(title = "Issue 'Created At' dates over time",
       subtitle = "Monthly counts annotated with a line indicating the 3-month simple moving average (SMA)", 
       y = "Issues Opened", 
       x = "") +
  scale_x_date(date_labels="%b %Y",date_breaks  ="3 month") +
  theme(axis.text.x = element_text(angle = 60, hjust = 1))

Graph presents counts of monthly certificates issued plus the simple moving average based on a 3-month window.
